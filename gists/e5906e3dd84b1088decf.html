Building a Go program via Docker

building-a-go-program-via-docker.sh

docker run -\
  -rm=true \
  -v $WORKSPACE/src:/gopath/src/github.com/foo/bar/src \
  -v $WORKSPACE/src:/app \
  -e "GOPATH=/gopath" \
  -w /app golang:1.5 sh \
  -c 'CGO_ENABLED=0 go build -a --installsuffix cgo --ldflags=\"-s\" -o bar'
