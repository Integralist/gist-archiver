Grid.css

/*
 * Algorithm taken from: 
 * http://csswizardry.com/2011/08/building-better-grid-systems/
 */

.row {
	width: (number of columns * width of one column) + (number of columns * width of one gutter) px;
	margin-left: -width of one gutter px;
	overflow: hidden;
	clear: both;
}

.grid {
	float: left;
	margin-left: width of one gutter px;
}
