package main

import (
	"fmt"
	"log"
	"net"
)

func main() {
	ipv4Addr, ipv4Net, err := net.ParseCIDR("10.0.0.0/8")
	if err != nil {
		log.Fatal(err)
	}
	fmt.Printf("%#v %s (%T)\n", ipv4Addr, ipv4Addr, ipv4Addr)
	fmt.Println(ipv4Net)
	fmt.Println(ipv4Net.Contains(net.ParseIP("10.255.255.0")))
}
