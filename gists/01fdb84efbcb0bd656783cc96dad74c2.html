Python Sort List by Dictionary Key

View original Gist on GitHub

Sort List by Dictionary Key.py

order = {
    1: 'foo',
    2: 'bar',
    3: 'baz'
}

def sort_by_key(i):
    for k, v in order.items():
        if i == v:
            return k

sorted(['bar', 'baz', 'foo'], key=sort_by_key) 

# ['foo', 'bar', 'baz']
