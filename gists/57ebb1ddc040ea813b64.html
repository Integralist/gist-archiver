Run Sass within Ruby.rb

require "sass"
options = {
  cache: true,
  syntax: :sass,
  style: :compressed,
  filename: original_file_name,
  ...
}
render = Sass::Engine.new(File.read(original_file_name), options).render
File.write(output_file_name, render)
