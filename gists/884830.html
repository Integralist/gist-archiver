Mobile Selector Engine.js

// A extremely basic selector engine for mobiles that support querySelectorAll /via @WebReflection (Andrea Giammarchi)
var $ = (function (s) {
    return function $(q) {
        return s.call(document.querySelectorAll(q))
    }
}([].slice));
