# generate config directory
mkdir ~/.config/torrench`

# download config file
curl -so ~/.config/torrench/config.ini https://pastebin.com/raw/reymRHSL`

# enable TBP searching
sed -i '' 's/\(enable = \)0/\11/' ~/.config/torrench/config.ini`

# search TBP
torrench -t 'ufc 219'`
