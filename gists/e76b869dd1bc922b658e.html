Ruby Sinatra Sessions

Ruby Sinatra Sessions.rb

require "sinatra"

enable :sessions
set :session_secret, "*&(^B234"

get "/" do
  id = get_session :foo
  "session id: #{id}\n"
end

def get_session(id)
  return session[id] unless session[id].nil?
  session[id] = SecureRandom.uuid
end
