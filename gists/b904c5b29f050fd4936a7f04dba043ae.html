// generate nanoseconds from microseconds

declare local var.start-nano INTEGER;
set var.start-nano = std.strtol(time.start.usec, 10);
set var.start-nano *= 1000;
log var.start-nano;
set req.http.X-Start-in-Nanoseconds = var.start-nano;
