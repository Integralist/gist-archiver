Go reverse slice loop.md

OLD SKOOL

package main

import (
	"fmt"
)

func main() {
	s := []int{1, 2, 3, 4, 5}
	for i := len(s) - 1; i >= 0; i-- {
		fmt.Println(s[i])
	}
}

MODERN

package main

import (
	"fmt"
	"slices"
)

func main() {
	s := []int{1, 2, 3, 4, 5}
	slices.Reverse(s)
	fmt.Println(s)
}
