def append_params(urls: list, params: str) -> list:
    """Returns copy of list with modified values.

    Example:
        append_params(["/foo", "/bar", "/baz"], "?id=123")

		-> ["/foo?id=123", "/bar?id=123", "/baz?id=123"]
    """
    return list(map(lambda v: f"{v}{params}", urls))
