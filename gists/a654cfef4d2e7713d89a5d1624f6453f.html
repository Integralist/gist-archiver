[Tiny Docker Builds]

View original Gist on GitHub

Tags: #go #golang #docker #build

Dockerfile

# build step
FROM golang:alpine AS build-env
ADD . /src
RUN cd /src && go build -o goapp

# final step
FROM alpine
WORKDIR /app
COPY --from=build-env /src/goapp /app/
ENTRYPOINT ./goapp
