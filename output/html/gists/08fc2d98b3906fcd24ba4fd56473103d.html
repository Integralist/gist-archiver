[Print the type of a reference] #rust #type #reflection

print type of reference.rs

use std::any::type_name;

fn type_of<T>(_: &T) {
    println!("{}", type_name::<T>())
}

fn main() {
    type_of(&"foo");               // &str
    type_of(&String::from("foo")); // alloc::string::String
    type_of(&123);                 // i32
    type_of(&1.23);                // f64
}
