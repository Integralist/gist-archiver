Jello::Config.rb

module Jello
  class Config
    class << self
      private

      def create_methods(items)
        items.each do |item|
          define_singleton_method item.downcase do
            @@config[item]
          end
        end
      end

      def get_config
        @@config ||= YAML.load(File.read "config/development.yaml")
      end
    end

    create_methods get_config.keys
  end
end
