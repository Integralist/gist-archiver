# Boilerplate for AMD, Node and browser global

## define.js

```javascript
(function(define) {
    'use strict';

    define(function() {
        return LIBRARY_NAME;
    });
}(typeof define == 'function' && define.amd
    ? define
    : function (factory) { typeof exports === 'object'
        ? (module.exports = factory())
        : (this.LIBRARY_NAME = factory());
    }
    // Boilerplate for AMD, Node, and browser global
));
```
