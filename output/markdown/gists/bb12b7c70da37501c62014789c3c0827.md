## Partial Application in Golang.go

```go
package main

import (
	"fmt"
)

func main() {
	crawl := crawl("logrus")
	crawl("other args")
}

type crawler func(string)

func crawl(logger string) crawler {
	return func(remaining string) {
		fmt.Println("logger:", logger, "remaining:", remaining)
	}
}

```
