# Bash Associative Arrays

## Bash Associative Arrays.sh

```shell
#!/bin/bash

declare -A arr

arr=(
  [foo1.foo.foo]=bar
  [baz]=qux
)

echo "${arr[foo1.foo.foo]}"

for item in "${arr[@]}"
do
  echo "value: $item"
done

declare -A arr2

arr2[foo]=bar
arr2[baz]=qux

for i in "${!arr2[@]}"  # access keys with !
do
  echo "key: $i"
  echo "value: ${arr2[$i]}"
done
```
