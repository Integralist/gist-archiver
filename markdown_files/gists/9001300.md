## 2. scrap.js

```javascript
var http = require('http');

function download(url, callback) {
    http.get(url, function(res) {
        var data = '';

        res.on('data', function(chunk) {
            data += chunk;
        });

        res.on('end', function() {
            callback(data);
        });
    }).on('error', function() {
        callback(null)
    });
}

module.exports = download;
```
